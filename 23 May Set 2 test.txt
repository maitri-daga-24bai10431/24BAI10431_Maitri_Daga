#q1-> List the Employee Name, their Department Name, and the Location Name for all employees
select employees.employee_name, departments.dept_name, departments.location_id from employees
join departments on employees.dept_id = departments.dept_id;

#q2-> Retrieve the Project Name, Client Name, and the Total Hours Worked on that project, but only for projects managed by the 'Consulting' department
select projects.project_name, projects.client_name, sum(assignments.hours_worked) from projects
join assignments on projects.project_id = assignments.project_id
join departments on projects.dept_id = departments.dept_id
where departments.dept_name = 'Consulting'
group by projects.project_name, projects.client_name;

#q3->Find the names of Employees who have worked on a project with Client Name 'GlobalTech Solutions'. List the Employee Name and Project Name

select employees.employee_name, projects.project_name from employees
join assignments on employees.employee_id = assignments.employee_id
join projects on assignments.project_id = projects.project_id
where projects.client_name = 'GlobalTech Solutions';

#q4-> List all Departments that have no Employees currently assigned to them
select departments.dept_name from departments
left join employees on departments.dept_id = employees.dept_id
where employees.employee_id is null;

#q5->Show the Manager Name, the Department Name they manage, and the total number of direct reports they have
select managers.employee_name, departments.dept_name, count(employees.employee_id) from employees
join employees managers on employees.manager_id = managers.employee_id
join departments on managers.dept_id = departments.dept_id
group by managers.employee_name, departments.dept_name;

#q6->Calculate the Average Hours Worked per Employee across all projects. List the Employee Name and the average hours
select employees.employee_name, avg(assignments.hours_worked) from employees
join assignments on employees.employee_id = assignments.employee_id
group by employees.employee_name;

#q7 -> Find the Department Name(s) that have an average salary above the company-wide average salary
select departments.dept_name from departments
join employees on departments.dept_id = employees.dept_id
group by departments.dept_name
having avg(employees.salary) > (
    select avg(salary) 
    from employees
);

#q8-> Find the Department Name(s) that have an average salary above the company-wide average salary
select projects.client_name from projects
join assignments on projects.project_id = assignments.project_id
group by projects.client_name
having sum(assignments.hours_worked) > 500;

#q10-> Calculate the total number of employees assigned to each Project, but only include projects that have more than 5 employees assigned
select projects.project_name, count(distinct assignments.employee_id) from projects
join assignments on projects.project_id = assignments.project_id
group by projects.project_name
having count(distinct (assignments.employee_id)) > 5;

#q12-> Retrieve the Department Name(s) that have employees assigned to more than one project
select departments.dept_name from departments
join employees on departments.dept_id = employees.dept_id
join assignments on employees.employee_id = assignments.employee_id
group by departments.dept_name
having count(distinct (assignments.project_id)) > 1;

#q13-> Write a query to list the Employee Name and Project Name for any employee who has worked less than 10 hours on a project
select employees.employee_name, projects.project_name from employees
join assignments on employees.employee_id = assignments.employee_id
join projects on assignments.project_id = projects.project_id
where assignments.hours_worked < 10;

#q14-> Find the Project Name(s) that have no assigned employees
select projects.project_name from projects
left join assignments on projects.project_id = assignments.project_id
where assignments.assignment_id is null;

#q15->Select the Employee Name(s) whose salary is greater than the maximum salary in their own department
select employees.employee_name from employees
where employees.salary > (
    select max(internal_employees.salary)
    from employees internal_employees
    where internal_employees.dept_id = employees.dept_id 
	and internal_employees.employee_id <> employees.employee_id
);

#q16-> Assuming the Assignments table has a start_date column, find the Employee Name and their oldest project start date
select employees.employee_name, min(assignments.start_date) from employees
join assignments on employees.employee_id = assignments.employee_id
group by employees.employee_name; 

#q19->. Find the top 3 Departments based on the total hours worked by all their employees across all projects
select departments.dept_name from departments
join employees on departments.dept_id = employees.dept_id
join assignments on employees.employee_id = assignments.employee_id
group by departments.dept_name
order by sum(assignments.hours_worked) desc
limit 3;

#q20->. Display the Project Name and a status ("High Activity" if total hours > 300, "Normal" otherwise) using the CASE statement

select projects.project_name, 
case 
	when sum(assignments.hours_worked) > 300 then 'High Activity'
	else 'Normal'
	end
from projects
join assignments on projects.project_id = assignments.project_id
group by projects.project_name;

#q21-> List the Employee Name and their Manager's Name, even if the employee has no manager, or if a potential manager has no employees reporting to them

select employees.employee_name, managers.employee_name from employees
left join employees managers on employees.manager_id = managers.employee_id
union
select employees.employee_name, managers.employee_name from employees
right join employees managers on employees.manager_id = managers.employee_id;

#q23->Find the Employee Name(s) who are assigned to the same projects as Employee ID 101
select distinct employees.employee_name from employees
join assignments on employees.employee_id = assignments.employee_id
where assignments.project_id in (
    select internal_assignments.project_id from assignments internal_assignments
    where internal_assignments.employee_id = 101
) and employees.employee_id <> 101;
