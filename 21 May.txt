# to select the 2nd max salary from every department 

SELECT first_name, department, salary
FROM worker w1
WHERE w1.salary = (
    SELECT DISTINCT w2.salary
    FROM worker w2
    WHERE w1.department = w2.department
    ORDER BY w2.salary DESC
    LIMIT 1 OFFSET 1
);
