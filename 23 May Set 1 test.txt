#q1-> List the name of each employee along with their job title and department name
select Employees.name, Jobs.job_title, Departments.dept_name
from Employees
join Jobs ON Employees.job_id = Jobs.job_id
join Departments ON Employees.dept_id = Departments.dept_id;

#q2 ->  Find all employees whose salary is higher than the average salary of the 'IT' department
select name, salary from Employees where salary > (
    select avg(salary)
    from Employees
    join Departments on Employees.dept_id = Departments.dept_id
    where Departments.dept_name = 'IT'
);

#q3 ->  Calculate the total salary expenditure for each department name
select Departments.dept_name,sum(salary) from Departments
join Employees on Departments.dept_id=Employees.dept_id
group by Departments.dept_name;

#q4-> Display departments that have more than 2 employees and an average salary above 70,000.
select Departments.dept_name from Departments
join employees on Departments.dept_id=Employees.dept_id
group by Departments.dept_name
having count(employees.emp_id)>2 and avg(salary)>70000;

#q6 -> Find the top 3 highest-paid employees and list them in descending order
Select name, salary from Employees order by salary desc limit 3;

#q7-> Find all Developers working in the 'San Francisco' location
select Employees.name, Jobs.job_title, Departments.location from Employees
join Jobs on Employees.job_id = Jobs.job_id
join Departments on Employees.dept_id = Departments.dept_id
where Jobs.job_title = 'Developer' and Departments.location = 'San Francisco';

#q8 -> Find the number of employees hired each year
select year(hire_date) ,count(emp_id) from employees group by year(hire_date);

#q9-> List job titles where the maximum salary of an employee in that role is less than 100,000
select jobs.job_title from jobs
join employees on jobs.job_id = employees.job_id
group by jobs.job_title
having max(employees.salary) < 100000;

#q18->Display employee names and hire dates, sorted by department name (A-Z) and then by salary (High to Low)
select employees.name, departments.dept_name, employees.salary, employees.hire_date from employees
join departments on employees.dept_id = departments.dept_id
order by departments.dept_name asc, employees.salary desc;

#q19->Identify managers (by ID) who manage more than 2 people

#q22->Find employees in the 'Finance' department whose names contain the letter 'i'

select employees.name, departments.dept_name from employees
join departments on employees.dept_id = departments.dept_id
where departments.dept_name = 'Finance' and employees.name like '%i%';

#q23-> Calculate the 10% bonus amount for each employee and show the total bonus per department
select departments.dept_name, sum(employees.salary * 0.10) from departments
join employees on departments.dept_id = employees.dept_id
group by departments.dept_name;

#q24->List the name of the department that pays the highest total salary
select departments.dept_name from departments
join employees on departments.dept_id = employees.dept_id
group by departments.dept_name
order by sum(employees.salary) desc limit 1;
